# Simple Python script to display Fibonacci sequence

def fibonacci(n):
    a, b = 0, 1
    result = []
    while len(result) < n:
        result.append(a)
        a, b = b, a + b
    return result

# Change the number 10 to how many Fibonacci numbers you want to generate
fib_sequence = fibonacci(10)
print("Fibonacci sequence:", fib_sequence)
